import React from 'react';

import sd2dItems from './items-7d2d.json';

export function SdtdItems() {
  return (
    <table>
      <thead>
        <tr>
          <th>Name</th>
          <th>Description</th>
          <th>Category</th>
          <th>Code</th>
        </tr>
      </thead>
      <tbody>
        {Object.values(sd2dItems).map((item) => {

          const description = String(item.description).split('\\n').map((line, index) => (
            <React.Fragment key={index}>
              {line}
              <br />
            </React.Fragment>
          ));

          return (
            <tr key={item.itemid}>
              <td>{item.name}</td>
              <td>{description}</td>
              <td>{item.group}</td>
              <td>{item.code}</td>
          </tr>)
        })}
      </tbody>
    </table>

);
}
